27class A2DPVolumeControl {
28
29 protected:
30 bool is_volume_used = false;
31 bool mono_downmix = false;
32 int32_t volumeFactor;
33 int32_t volumeFactorMax;
34
35 public:
36 A2DPVolumeControl() {
37 volumeFactorMax = 0x1000;
38 }
39
40 virtual void update_audio_data(Frame* data, uint16_t frameCount) {
41 if (data!=nullptr && frameCount>0 && ( mono_downmix || is_volume_used)) {
42 ESP_LOGD("VolumeControl", "update_audio_data");
43 for (int i=0;i<frameCount;i++){
44 int32_t pcmLeft = data[i].channel1;
45 int32_t pcmRight = data[i].channel2;
46 // if mono -> we provide the same output on both channels
47 if (mono_downmix) {
48 pcmRight = pcmLeft = (pcmLeft + pcmRight) / 2;
49 }
50 // adjust the volume
51 if (is_volume_used) {
52 pcmLeft = pcmLeft * volumeFactor / volumeFactorMax;
53 pcmRight = pcmRight * volumeFactor / volumeFactorMax;
54 }
55 data[i].channel1 = pcmLeft;
56 data[i].channel2 = pcmRight;
57 }
58 }
59 }
60
61 // provides a factor in the range of 0 to 4096
62 int32_t get_volume_factor() {
63 return volumeFactor;
64 }
65
66 // provides the max factor value 4096
67 int32_t get_volume_factor_max() {
68 return volumeFactorMax;
69 }
70
71 void set_enabled(bool enabled) {
72 is_volume_used = enabled;
73 }
74
75 void set_mono_downmix(bool enabled) {
76 mono_downmix = enabled;
77 }
78
79 virtual void set_volume(uint8_t volume) = 0;
80};

87class A2DPDefaultVolumeControl : public A2DPVolumeControl {
88
89 virtual void set_volume(uint8_t volume) {
90 constexpr double base = 1.4;
91 constexpr double bits = 12;
92 constexpr double zero_ofs = pow(base, -bits);
93 constexpr double scale = pow(2.0, bits);
94 double volumeFactorFloat = (pow(base, volume * bits / 127.0 - bits) - zero_ofs) * scale / (1.0 - zero_ofs);
95 volumeFactor = volumeFactorFloat;
96 if (volumeFactor > 0x1000) {
97 volumeFactor = 0x1000;
98 }
99 }
100};

106class A2DPSimpleExponentialVolumeControl : public A2DPVolumeControl {
107 virtual void set_volume(uint8_t volume) {
108 double volumeFactorFloat = volume;
109 volumeFactorFloat = pow(2.0, volumeFactorFloat * 12.0 / 127.0);
110 int32_t volumeFactor = volumeFactorFloat - 1.0;
111 if (volumeFactor > 0xfff) {
112 volumeFactor = 0xfff;
113 }
114 }
