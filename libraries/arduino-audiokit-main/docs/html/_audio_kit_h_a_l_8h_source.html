82struct AudioKitConfig {
83 i2s_port_t i2s_num = (i2s_port_t)0;
84#if defined(ESP32) && ESP_IDF_VERSION_MAJOR < 4
85 int mclk_gpio = 0; // default value
86#else
87 int mclk_gpio = -1; // take definition from board_pins_config.c
88#endif
89 bool sd_active = true;
90 bool auto_clear = true;
91 bool use_apll = true;
92 bool i2s_active = true;
93 int buffer_count = 6;
94 int buffer_size = 512;
95
96 audio_hal_adc_input_t adc_input = AUDIOKIT_DEFAULT_INPUT; /*!< set adc channel with audio_hal_adc_input_t*/
97 audio_hal_dac_output_t dac_output =AUDIOKIT_DEFAULT_OUTPUT; /*!< set dac channel */
98 audio_hal_codec_mode_t codec_mode; /*!< select codec mode: adc, dac or both */
99 audio_hal_iface_mode_t master_slave_mode = AUDIOKIT_DEFAULT_MASTER_SLAVE; /*!< audio codec chip mode */
100 audio_hal_iface_format_t fmt = AUDIOKIT_DEFAULT_I2S_FMT; /*!< I2S interface format */
101 audio_hal_iface_samples_t sample_rate = AUDIOKIT_DEFAULT_RATE; /*!< I2S interface samples per second */
102 audio_hal_iface_bits_t bits_per_sample = AUDIOKIT_DEFAULT_BITSIZE; /*!< i2s interface number of bits per sample */
103
104 /// Returns true if the CODEC is the master
105 bool isMaster() { return master_slave_mode == AUDIO_HAL_MODE_MASTER; }
106
107 /// provides the bits per sample
108 int bitsPerSample() {
109 switch (bits_per_sample) {
110 case AUDIO_HAL_BIT_LENGTH_16BITS:
111 return 16;
112 case AUDIO_HAL_BIT_LENGTH_24BITS:
113 return 24;
114 case AUDIO_HAL_BIT_LENGTH_32BITS:
115 return 32;
116 }
117 KIT_LOGE("bits_per_sample not supported: %d", bits_per_sample);
118 return 0;
119 }
120
121 /// Provides the sample rate in samples per second
122 uint32_t sampleRate() {
123 switch (sample_rate) {
124 case AUDIO_HAL_08K_SAMPLES: /*!< set to 8k samples per second */
125 return 8000;
126 case AUDIO_HAL_11K_SAMPLES: /*!< set to 11.025k samples per second */
127 return 11025;
128 case AUDIO_HAL_16K_SAMPLES: /*!< set to 16k samples in per second */
129 return 16000;
130 case AUDIO_HAL_22K_SAMPLES: /*!< set to 22.050k samples per second */
131 return 22050;
132 case AUDIO_HAL_24K_SAMPLES: /*!< set to 24k samples in per second */
133 return 24000;
134 case AUDIO_HAL_32K_SAMPLES: /*!< set to 32k samples in per second */
135 return 32000;
136 case AUDIO_HAL_44K_SAMPLES: /*!< set to 44.1k samples per second */
137 return 44100;
138 case AUDIO_HAL_48K_SAMPLES: /*!< set to 48k samples per second */
139 return 48000;
140 }
141 KIT_LOGE("sample rate not supported: %d", sample_rate);
142 return 0;
143 }

145#if AUDIOKIT_ESP32_I2S && defined(ESP32)
146 /// Provides the ESP32 i2s_config_t to configure I2S
147 i2s_config_t i2sConfig() {
148 // use just the oposite of the CODEC setting
149 const i2s_config_t i2s_config = {
150 .mode = i2sMode(),
151 .sample_rate = sampleRate(),
152 .bits_per_sample = (i2s_bits_per_sample_t)bitsPerSample(),
153 .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,
154 .communication_format = (i2s_comm_format_t)i2sFormat(),
155 .intr_alloc_flags = ESP_INTR_FLAG_LEVEL2 | ESP_INTR_FLAG_IRAM,
156 .dma_buf_count = buffer_count,
157 .dma_buf_len = buffer_size,
158 .use_apll = use_apll,
159 .tx_desc_auto_clear = auto_clear,
160 };
161 return i2s_config;
162 }
163
164 /// Provides the ESP32 i2s_pin_config_t to configure the pins for I2S
165 i2s_pin_config_t i2sPins() {
166 i2s_pin_config_t result;
167 get_i2s_pins(i2s_num, &result);
168#if ESP_IDF_VERSION > ESP_IDF_VERSION_VAL(4, 4, 0)
169 // overwrite mclk from board def if necessary
170 if (mclk_gpio>=0){
171 result.mck_io_num = (gpio_num_t)mclk_gpio; //mclk_gpio;
172 }
173#endif
174 return result;
175 }

177 protected:
178 i2s_comm_format_t i2sFormat(){
179 i2s_comm_format_t its_com_fmt = (i2s_comm_format_t) I2S_COMM_FORMAT_STAND_I2S;
180 if (fmt==AUDIO_HAL_I2S_LEFT){
181 its_com_fmt = (i2s_comm_format_t) I2S_COMM_FORMAT_STAND_MSB;
182 } else if(fmt==AUDIO_HAL_I2S_RIGHT){
183 its_com_fmt = (i2s_comm_format_t) I2S_COMM_FORMAT_STAND_MSB;
184 } else if(fmt==AUDIO_HAL_I2S_DSP){
185 its_com_fmt = (i2s_comm_format_t )I2S_COMM_FORMAT_STAND_PCM_SHORT;
186 }
187 return its_com_fmt;
188 }
189
190 i2s_mode_t i2sMode() {
191 int mode = isMaster() ? I2S_MODE_SLAVE : I2S_MODE_MASTER;
192 // using ESP32 dac/adc
193 if (fmt == AUDIO_HAL_I2S_DSP){
194 #if AUDIOKIT_BOARD==4 || defined(CONFIG_IDF_TARGET_ESP32S3) || defined(CONFIG_IDF_TARGET_ESP32S2)|| defined(CONFIG_IDF_TARGET_ESP32C3)
195 KIT_LOGE("AUDIO_HAL_I2S_DSP not supported");
196 #else
197 if (codec_mode == AUDIO_HAL_CODEC_MODE_DECODE) {
198 mode = mode | I2S_MODE_TX | I2S_MODE_DAC_BUILT_IN;
199 } else if (codec_mode == AUDIO_HAL_CODEC_MODE_ENCODE) {
200 mode = mode | I2S_MODE_RX | I2S_MODE_ADC_BUILT_IN;
201 } else if (codec_mode == AUDIO_HAL_CODEC_MODE_BOTH) {
202 mode = mode | I2S_MODE_RX | I2S_MODE_TX | I2S_MODE_ADC_BUILT_IN | I2S_MODE_DAC_BUILT_IN;
203 }
204 #endif
205 } else {
206 // I2S
207 if (codec_mode == AUDIO_HAL_CODEC_MODE_DECODE) {
208 mode = mode | I2S_MODE_TX;
209 } else if (codec_mode == AUDIO_HAL_CODEC_MODE_ENCODE) {
210 mode = mode | I2S_MODE_RX;
211 } else if (codec_mode == AUDIO_HAL_CODEC_MODE_BOTH) {
212 mode = mode | I2S_MODE_RX | I2S_MODE_TX;
213 }
214 }
215 return (i2s_mode_t) mode;
216 }
217
218#endif
219};

261 /// Starts the codec
262 bool begin(AudioKitConfig cnfg) {
263 KIT_LOGI(LOG_METHOD);
264 cfg = cnfg;
265 KIT_LOGI("Selected board: %d", AUDIOKIT_BOARD);
266
267 // if already active we stop first
268 if (is_active){
269 end();
270 }
271
272#if AUDIOKIT_SETUP_SD
273 // release SPI for SD card if it is not necessary
274 if (AUDIOKIT_SETUP_SD && !cfg.sd_active){
275 p_spi->end();
276 }
277#endif
278 // setup headphone if necessary
279 setupHeadphoneDetection();
280
281 audio_hal_conf.adc_input = cfg.adc_input;
282 audio_hal_conf.dac_output = cfg.dac_output;
283 audio_hal_conf.codec_mode = cfg.codec_mode;
284 audio_hal_conf.i2s_iface.mode = cfg.master_slave_mode;
285 audio_hal_conf.i2s_iface.fmt = cfg.fmt;
286 audio_hal_conf.i2s_iface.samples = cfg.sample_rate;
287 audio_hal_conf.i2s_iface.bits = cfg.bits_per_sample;
288
289 // init HAL
290 if (hal_handle == 0) {
291 hal_handle = audio_hal_init(&audio_hal_conf, &AUDIO_DRIVER);
292 if (hal_handle == 0) {
293 KIT_LOGE("audio_hal_init");
294 return false;
295 }
296 }
297
298 if (cfg.i2s_active){
299 setupI2S(cfg);
300 }
301
302 // call start
303 if (!setActive(true)) {
304 KIT_LOGE("setActive");
305 return false;
306 }
307
308 // activate speaker when output not earphones only
309 setSpeakerActive(cfg.dac_output!=AUDIO_HAL_DAC_OUTPUT_LINE1);
310
311 is_active = true;
312 return true;
313 }
314
315 /// Stops the CODEC
316 bool end() {
317 KIT_LOGI(LOG_METHOD);
318
319#if AUDIOKIT_ESP32_I2S && defined(ESP32)
320 if (cfg.i2s_active){
321 // uninstall i2s driver
322 i2s_driver_uninstall(cfg.i2s_num);
323 }
324#endif
325 // stop codec driver
326 audio_hal_ctrl_codec(hal_handle, cfg.codec_mode, AUDIO_HAL_CTRL_STOP);
327 // deinit
328 audio_hal_deinit(hal_handle);
329 // this will save power
330 setSpeakerActive(false);
331
332 is_active = false;
333 hal_handle = 0;
334 return true;
335 }

338 AudioKitConfig config() { return cfg; }

340 /// Sets the codec active / inactive

343 }

565 /**
566 * @brief Switch off the PA if the headphone in plugged in
567 * and switch it on again if the headphone is unplugged
568 *
569 */
570 static void actionHeadphoneDetection() {
571 if (selfAudioKit->headphonePin>0){
572
573 // detect changes
574 bool isConnected = selfAudioKit->headphoneStatus();
575 if (selfAudioKit->headphoneIsConnected != isConnected) {
576 selfAudioKit->headphoneIsConnected = isConnected;
577
578 // update if things have stabilized
579 bool powerActive = !isConnected;
580 KIT_LOGW("Headphone jack has been %s", isConnected ? "inserted" : "removed");
581 selfAudioKit->setSpeakerActive(powerActive);
582 }
583 }
584 yield();
585 }

660bool setupI2S(AudioKitConfig cnfg) {
661#if AUDIOKIT_ESP32_I2S && defined(ESP32)
662 KIT_LOGD("setupI2S");
663 // setup i2s driver - with no queue
664 i2s_config_t i2s_config = cfg.i2sConfig();
665 if (i2s_driver_install(cfg.i2s_num, &i2s_config, 0, NULL) != ESP_OK) {

670 // define i2s pins

672 KIT_LOGI("i2s_set_pin");
673
674 KIT_LOGI("- bck_io_num: %d", pin_config.bck_io_num);
675 KIT_LOGI("- ws_io_num: %d", pin_config.ws_io_num);
676 KIT_LOGI("- data_out_num: %d", pin_config.data_out_num);
677 KIT_LOGI("- data_in_num: %d", pin_config.data_in_num);
678# if ESP_IDF_VERSION > ESP_IDF_VERSION_VAL(4, 4, 0)
679 KIT_LOGI("- mck_io_num: %d", pin_config.mck_io_num);
680# endif
